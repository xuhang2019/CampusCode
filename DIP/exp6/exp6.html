def get_Q(img):
    Q = np.zeros([256,256])
    m,n = img.shape
    for i in range(m):
        for j in range(n):
            if j == n-1:
                continue
            left = img[i,j]
            right = img[i,j+1]
            Q[left,right] += 1
    return Q

def get_msz(Q):
    m,n = Q.shape
    Q21 = Q/np.sum(Q)
    p_max= np.max(Q21)
    rel = 0
    cont = 0
    unif = 0
    homo = 0
    s = 0
    
    mr = 0
    mc = 0
    for i in range(m):
        for j in range(n):
            mr += i*Q21[i,j]
            mc += j*Q21[i,j]
    sigmar = 0
    sigmac = 0
    for i in range(m):
        for j in range(n):
            sigmar += (i-mr)**2*Q21[i,j]
            sigmac += (j-mc)**2*Q21[i,j]
    sigmar = sigmar ** 0.5
    sigmac = sigmac ** 0.5
    
    for i in range(m):
        for j in range(n):
            rel += (i-mr)*(j-mc)*Q21[i,j]/(sigmar*sigmac)
            cont += (i-j)**2*Q21[i,j]
            unif += Q21[i,j]**2
            homo += Q21[i,j]/(1+np.abs(i-j))
            if Q21[i,j] != 0:
                s  += Q21[i,j] * np.log2(Q21[i,j])
            
    msz = [p_max,rel,cont,unif,homo,-s]
    return msz
